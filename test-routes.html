// Agregado: Cache para prevenir re-renders infinitos
let cachedSnapshot = null;

function getSnapshot() {
  const newNotifications = list();
  const newUnread = getUnreadCount();
  const newQuietHours = getQuietHoursState();

  // Solo crear nuevo objeto si los datos realmente cambiaron
  if (
    !cachedSnapshot ||
    notifications !== cachedSnapshot._rawNotifications ||
    cachedSnapshot.unread !== newUnread ||
    quietHours !== cachedSnapshot._rawQuietHours
  ) {
    cachedSnapshot = {
      notifications: newNotifications,
      unread: newUnread,
      quietHours: newQuietHours,
      _rawNotifications: notifications,
      _rawQuietHours: quietHours,
    };
  }

  return cachedSnapshot;
}

// Modificado: emit() invalida el cache
function emit() {
  cachedSnapshot = null; // ‚Üê Nuevo
  listeners.forEach((l) => l());
}

export function useNotificationStore() {
